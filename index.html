import requests from bs4 import BeautifulSoup URL = "https://www.billboard-japan.com/charts/detail?a=hot100" headers = { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64)AppleWebKit/537.36 (KHTML, like Gecko) Chrome/73.0.3683.86 Safari/537.36'} data = requests.get(URL, headers=headers) soup = BeautifulSoup(data.text, 'html.parser') dele = soup.find('td', class_='exbanner').decompose() billbords = soup.select('#content2 > div > div.leftBox > table > tbody > tr') for billbord in billbords: rank = billbord.select_one('span').text title = billbord.select_one('.musuc_title').text.strip() artist = billbord.select_one('.artist_name').text.strip() print(rank, title, artist)
